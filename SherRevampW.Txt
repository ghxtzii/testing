-- Airshot detection and AimPart usage
function IsAirborne(target)
    local hrp = target:FindFirstChild("HumanoidRootPart")
    if not hrp then return false end
    local ray = Ray.new(hrp.Position, Vector3.new(0, -5, 0))
    local part = workspace:FindPartOnRay(ray, target)
    return part == nil
end

-- Get selected AimPart or AirPart
function GetTargetPart(target)
    local airPart = getgenv().setting.airpart
    local normalPart = getgenv().setting.aimpart
    local useAir = IsAirborne(target)
    local partName = useAir and airPart or normalPart
    return target:FindFirstChild(partName)
end

-- Prediction logic
function GetPredictedPosition(target)
    local hrp = target:FindFirstChild("HumanoidRootPart")
    if not hrp then return nil end
    local velocity = hrp.Velocity
    local pred = getgenv().setting.prediction.prediction
    return hrp.Position + (velocity * pred)
end

-- Camlock Aim logic
function CamlockTo(target)
    local part = GetTargetPart(target)
    if part then
        local predicted = part.Position
        if getgenv().setting.camlock.smoothenabled then
            Camera.CFrame = Camera.CFrame:Lerp(CFrame.new(Camera.CFrame.Position, predicted), getgenv().setting.camlock.smoothvalue / 10)
        else
            Camera.CFrame = CFrame.new(Camera.CFrame.Position, predicted)
        end
    end
end

-- __namecall Hook with Anti AimViewer
local mt = getrawmetatable(game)
setreadonly(mt, false)
local old = mt.__namecall

mt.__namecall = newcclosure(function(Self, ...)
    local args = {...}
    local method = getnamecallmethod()
    local Target = rawget(getgenv(), "CurrentTarget")

    if method == "FireServer" and typeof(args[1]) == "Vector3" then
        if Target and not getgenv().setting.Toggles.AntiAimViewer then
            args[1] = GetPredictedPosition(Target)
            return old(Self, unpack(args))
        end
    elseif method == "FireServer" and typeof(args[1]) == "table" then
        for i,v in ipairs(args[1]) do
            if typeof(v) == "Vector3" and Target and not getgenv().setting.Toggles.AntiAimViewer then
                args[1][i] = GetPredictedPosition(Target)
            end
        end
        return old(Self, unpack(args))
    end

    return old(Self, ...)
end)

local Drawing = Drawing or {} -- fallback for compatibility
local espCache = {}

function CreateESP(plr)
    local box = Drawing.new("Square")
    box.Thickness = 2
    box.Transparency = 1
    box.Filled = false
    box.Visible = false

    local name = Drawing.new("Text")
    name.Size = getgenv().setting.ESP.Font_Size
    name.Center = true
    name.Outline = true
    name.Font = getgenv().setting.ESP.Font
    name.Visible = false

    local tracer = Drawing.new("Line")
    tracer.Thickness = 1.5
    tracer.Transparency = 1
    tracer.Visible = false

    espCache[plr] = {Box = box, Name = name, Tracer = tracer}
end

function UpdateESP(plr, char)
    local hrp = char:FindFirstChild("HumanoidRootPart")
    if not hrp then return end

    local pos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
    local visible = true
    local color = visible and Color3.fromRGB(0, 255, 0) or Color3.fromRGB(255, 0, 0)

    local esp = espCache[plr]
    if not esp then return end

    esp.Box.Position = Vector2.new(pos.X - 25, pos.Y - 50)
    esp.Box.Size = Vector2.new(50, 100)
    esp.Box.Color = color
    esp.Box.Visible = onScreen and getgenv().setting.ESP.BoxEsp

    esp.Name.Position = Vector2.new(pos.X, pos.Y - 60)
    esp.Name.Text = plr.DisplayName
    esp.Name.Color = getgenv().setting.ESP.Name_Color
    esp.Name.Visible = onScreen and getgenv().setting.ESP.NameEsp

    esp.Tracer.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y)
    esp.Tracer.To = Vector2.new(pos.X, pos.Y)
    esp.Tracer.Color = color
    esp.Tracer.Visible = onScreen and getgenv().setting.ESP.VisibleCheck
end

function PlayHitEffect()
    local effectName = getgenv().setting.Visuals.HitEffect
    local target = rawget(getgenv(), "CurrentTarget")
    if not target then return end

    local effectAttachment = ReplicatedStorage:FindFirstChild(effectName, true)
    if effectAttachment and target:FindFirstChild("HumanoidRootPart") then
        local clone = effectAttachment:Clone()
        clone.Parent = target.HumanoidRootPart

        for _, p in pairs(clone:GetChildren()) do
            if p:IsA("ParticleEmitter") then
                p.Enabled = true
                delay(0.5, function() p.Enabled = false end)
            end
        end

        game:GetService("Debris"):AddItem(clone, 2)
    end
end

-- Body Chams
local function ApplyChams(plr)
    if not plr.Character then return end
    for _, part in pairs(plr.Character:GetChildren()) do
        if part:IsA("BasePart") then
            part.Material = getgenv().setting.Visuals.BodyCham_Material
            part.Color = getgenv().setting.Visuals.BodyCham_Color
        end
    end
end

-- Example: Apply on current target
RunService.RenderStepped:Connect(function()
    local tgt = rawget(getgenv(), "CurrentTarget")
    if tgt then
        ApplyChams(tgt)
    end
end)

local Window = Library:CreateWindow("Sher Revamp", Vector2.new(500, 600), Enum.KeyCode.RightControl)

-- Tabs
local Main = Window:CreateTab("Main")
local Visuals = Window:CreateTab("Visuals")
local ESP = Window:CreateTab("ESP")

-- Main Toggles
Main:CreateToggle("Camlock", getgenv().setting.Toggles.Camlock, function(v)
    ToggleCamlock(v)
end)
Main:CreateToggle("Spectate On", getgenv().setting.Toggles.Spectate, function(v)
    ToggleSpectate(v)
end)
Main:CreateToggle("View On", getgenv().setting.Toggles.ViewOn, function(v)
    ToggleViewOn(v)
end)
Main:CreateToggle("Anti AimViewer", getgenv().setting.Toggles.AntiAimViewer, function(v)
    ToggleAntiAimViewer(v)
end)

Main:CreateToggle("Prediction Breaker", getgenv().setting.Toggles.PredBreaker, function(v)
    TogglePredBreaker(v, getgenv().setting.Toggles.PredBreaker_Amount)
end)

Main:CreateSlider("Prediction Amount", 0, 1, getgenv().setting.prediction.prediction, function(val)
    getgenv().setting.prediction.prediction = val
end)

Main:CreateSlider("Smoothness", 0, 10, getgenv().setting.camlock.smoothvalue, function(val)
    getgenv().setting.camlock.smoothvalue = val
end)

-- Visuals Tab
Visuals:CreateDropdown("Hit Effect", {"Crescent Slash", "Cosmic Explosion", "Coom", "Slash", "Atomic Slash", "Nova Impact"}, function(val)
    getgenv().setting.Visuals.HitEffect = val
end)

Visuals:CreateDropdown("Hit Sound", {"RIFK7", "Skeet", "Fatality", "UwU", "TF2", "Gamesense", "Click", "Bonk"}, function(val)
    getgenv().setting.Visuals.HitSound = val
end)

Visuals:CreateDropdown("Body Cham Material", {"ForceField", "Plastic", "Neon", "Glass"}, function(val)
    getgenv().setting.Visuals.BodyCham_Material = Enum.Material[val]
end)

Visuals:CreateColorPicker("Cham Color", getgenv().setting.Visuals.BodyCham_Color, function(val)
    getgenv().setting.Visuals.BodyCham_Color = val
end)

Visuals:CreateToggle("Crosshair Animate", getgenv().setting.Visuals.Crosshair_Animation, function(val)
    getgenv().setting.Visuals.Crosshair_Animation = val
end)

-- ESP Tab
ESP:CreateToggle("Box ESP", getgenv().setting.ESP.BoxEsp, function(val)
    getgenv().setting.ESP.BoxEsp = val
end)

ESP:CreateToggle("Name ESP", getgenv().setting.ESP.NameEsp, function(val)
    getgenv().setting.ESP.NameEsp = val
end)

ESP:CreateToggle("Tracers", getgenv().setting.ESP.VisibleCheck, function(val)
    getgenv().setting.ESP.VisibleCheck = val
end)

ESP:CreateColorPicker("Visible Color", Color3.fromRGB(0,255,0), function(val)
    -- Hook visibility coloring logic here
end)

ESP:CreateColorPicker("Invisible Color", Color3.fromRGB(255,0,0), function(val)
    -- Hook invis logic here
end)

-- UI Toggle Button Box - Open/Close UI
local ScreenGui = Instance.new("ScreenGui", game:GetService("CoreGui"))
ScreenGui.Name = "SherWOpener"
ScreenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
ScreenGui.ResetOnSpawn = false

local ToggleFrame = Instance.new("Frame", ScreenGui)
ToggleFrame.Size = UDim2.new(0, 100, 0, 35)
ToggleFrame.Position = UDim2.new(0, 10, 0.5, -50)
ToggleFrame.BackgroundColor3 = Color3.fromRGB(40, 0, 60)
ToggleFrame.BorderSizePixel = 0
ToggleFrame.Active = true
ToggleFrame.Draggable = true

local UICorner = Instance.new("UICorner", ToggleFrame)
UICorner.CornerRadius = UDim.new(0, 8)

local ToggleButton = Instance.new("TextButton", ToggleFrame)
ToggleButton.Size = UDim2.new(1, 0, 1, 0)
ToggleButton.BackgroundTransparency = 1
ToggleButton.Text = "SherW"
ToggleButton.TextColor3 = Color3.fromRGB(255, 255, 255)
ToggleButton.Font = Enum.Font.GothamBold
ToggleButton.TextScaled = true

-- Reference to Library UI
local LibraryUI = game:GetService("CoreGui"):FindFirstChild("Sher Revamp")

ToggleButton.MouseButton1Click:Connect(function()
    if LibraryUI then
        LibraryUI.Enabled = not LibraryUI.Enabled
    end
end)
